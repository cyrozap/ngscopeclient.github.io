449
450 //Set the new trigger and sync to hardware
451 m_trigger = trigger;
452 PushTrigger();
453
454 //Delete old trigger *after* pushing the new one.
455 //This prevents possible race conditions where we disable the current trigger channel before the new
456 //trigger is set.
457 if(old_trig != trigger)
458 delete old_trig;
459 }
