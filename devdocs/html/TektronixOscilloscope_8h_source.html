320 bool IsAnalog(size_t index)
321 { return index < m_analogChannelCount; }
322
323 bool IsDigital(size_t index)
324 {
325 if(index < m_digitalChannelBase)
326 return false;
327 if(index >= (m_digitalChannelBase + 8*m_analogChannelCount) )
328 return false;
329 return true;
330 }
331
332 bool IsSpectrum(size_t index)
333 {
334 if(index < m_spectrumChannelBase)
335 return false;
336 if(index >= (m_spectrumChannelBase + m_analogChannelCount) )
337 return false;
338 return true;
339 }
